# functions for analysis
chiFilter <- function(c1, c2){
  # c1 = first column of interest
  # c2 = second column of interest
  df1 <- data.frame(c1, c2) %>% na.omit()
  tab <- table(df1$c1, df1$c2)
  print(tab)
  chi <- chisq.test(tab)
  return(chi)
}
